#include <stdio.h>
#include <time.h>

// Iterative function for factorial
long long factorial_iterative(int n) {
    long long result = 1;
    for (int i = 1; i <= n; i++)
        result *= i;
    return result;
}

// Recursive function for factorial
long long factorial_recursive(int n) {
    if (n == 0 || n == 1)
        return 1;
    else
        return n * factorial_recursive(n - 1);
}

int main() {
    int n;
    clock_t start, end;
    double cpu_time_used;
    long long result;

    printf("Enter a number to find factorial: ");
    scanf("%d", &n);

    // Iterative function timing
    start = clock();
    result = factorial_iterative(n);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("\n[Iterative] Factorial of %d = %lld", n, result);
    printf("\n[Iterative] Time taken: %f seconds", cpu_time_used);
    printf("\n[Iterative] Space Complexity ≈ O(1)\n");

    // Recursive function timing
    start = clock();
    result = factorial_recursive(n);
    end = clock();
    cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
    printf("\n[Recursive] Factorial of %d = %lld", n, result);
    printf("\n[Recursive] Time taken: %f seconds", cpu_time_used);
    printf("\n[Recursive] Space Complexity ≈ O(n) (due to recursion stack)\n");

    return 0;
}

#include <stdio.h>
#include <time.h>

int bs(int a[], int l, int h, int x)
{
    if(l > h) return -1;

    int m = (l + h) / 2;

    if(a[m] == x) return m;
    if(x < a[m]) return bs(a, l, m - 1, x);
    return bs(a, m + 1, h, x);
}

int main()
{
    int a[100], n, x, r;
    clock_t s, e;
    double t;

    printf("Enter n: ");
    scanf("%d", &n);

    printf("Enter sorted elements:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &a[i]);

    printf("Enter key: ");
    scanf("%d", &x);

    s = clock();
    r = bs(a, 0, n - 1, x);
    e = clock();

    t = (double)(e - s) / CLOCKS_PER_SEC;

    if(r == -1)
        printf("Not Found\n");
    else
        printf("Found at position %d\n", r + 1);

    printf("Time = %.10f sec\n", t);
    printf("Time Complexity = O(log n)");
    
    return 0;
}

#include<stdio.h>

int main()
{
    int a[100], n, i, j, t;

    printf("Enter n: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
        scanf("%d",&a[i]);

    for(i=0;i<n-1;i++)
    {
        for(j=0;j<n-i-1;j++)
        {
            if(a[j] > a[j+1])
            {
                t=a[j];
                a[j]=a[j+1];
                a[j+1]=t;
            }
        }
    }

    printf("Sorted Array:\n");

    for(i=0;i<n;i++)
        printf("%d ",a[i]);

    return 0;
}
#include<stdio.h>

int main()
{
    int a[100], n, i, j, key;

    printf("Enter n: ");
    scanf("%d",&n);

    for(i=0;i<n;i++)
        scanf("%d",&a[i]);

    for(i=1;i<n;i++)
    {
        key = a[i];
        j = i-1;

        while(j>=0 && a[j] > key)
        {
            a[j+1] = a[j];
            j--;
        }

        a[j+1] = key;
    }

    printf("Sorted Array:\n");

    for(i=0;i<n;i++)
        printf("%d ",a[i]);

    return 0;
}
